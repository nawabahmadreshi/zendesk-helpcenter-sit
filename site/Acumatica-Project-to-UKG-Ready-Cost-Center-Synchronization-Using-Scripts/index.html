Acumatica Project to UKG Ready Cost Center Synchronization Using Scripts

← Back to index

This guide covers the end-to-end tasks in setting up a script synchronization to sync the Project resource of Acumatica to the Cost Center resource of UKG Ready.

The following tasks are involved in setting up the synchronization from Acumatica to UKG Ready:

  • Initial Preparation
  • Configure Applications in Aquera Admin Portal
  • Synchronization Setup
  • Run Script Synchronization

Initial Preparation

Aquera Admin Portal

  1. Your IT admin must register as an Aquera Admin.
  2. Once registered successfully, Aquera creates a tenant for the IT Admin and provides access to it.

Configure Applications in Aquera Admin Portal

You must add and configure the following applications in the Aquera Admin portal:

  • Acumatica: Refer to the Acumatica Configuration Guide for details.
  • UKG Ready: Refer to the UKG Ready Configuration Guide for details.

Synchronization Setup

This section details how the synchronization must be configured to allow data push from Acumatica to UKG Ready.

Create Script Synchronization

Perform the following steps to create a new script synchronization:

  1. Log in to the Aquera Admin portal.
  2. Navigate to Synchronizations on the left panel which displays the list of synchronizations created by the user. 
    Note: The number of synchronization records displayed on this page is based on the display size of the user's screen.
  3. Click Create Synchronization.
  4. Select Script.
  5. Enter a name and a description for the synchronization.
  6. Click Continue. A new page is displayed to configure the synchronization.
  7. Click Create to create the script synchronization.

Configure Script Synchronization

This document describes how to set up the synchronization using the Script Synchronization method.

Connections

  • Set Acumatica as an application from the Connections tab with the variable acumatica.
  • Set UKG Ready as an application from the Connections tab with the variable ukg.

Refer to the Synchronization Setup Using Scripts and Working With Synchronization Scripts guides for details on the configurations.

Script

This section allows you to write a script to set up the synchronization. The script is based on JavaScript, so all the functions and language constructs are available in the script.

Note: The following script covers the synchronization of the Project resource type from Acumatica to the Cost Center resource type of UKG Ready. During the synchronization, the script checks the existence of the Acumatica Project in UKG Ready based on the cost center level name and creates a new cost center in UKG Ready if a matching cost center does not exist. You must modify the script based on your use case.

Configure the following script in the Script tab to synchronize the Project resource of Acumatica to the Cost Center resource of UKG Ready:

let ukgCC = ukg.getCostCenters();
let reportOnly = false;
let logsCreated = "Name,Level,ParentID,ExternalID,GlCode\n";
let logsFailed = "Name,Level,ErrorDetails\n";
let logsIgnored = "Name,Level,ParentID,ExternalID,GlCode\n";
let context = {};
context['$filter'] = "ProjectID eq '23THDRR028'"
context['$custom'] = "Project.ContractID";
let projectsObj = acumatica.getProject(context);
let projects = JSON.parse(projectsObj.body);
for(var project of projects){
    let level1Name = project.ProjectID.value+"--"+project.Description.value;
    //Find if the level 1 CC Already exists
    let level1Found = ukgCC.find(cc = cc.name === level1Name) != undefined ? true : false;
    let level1ID = null;
    let level1Obj = null;
    let level1ExternalId = project.custom.Project.ContractID.value/*Add the externalID*/;
    if(level1Found == true){
        log.info("Level 1 Cost center "+level1Name+" Hence not created");
        level1Obj = ukgCC.find(cc = cc.name === level1Name);
        level1ID = level1Obj.id;
        logsIgnored+= level1Name+","+1+","+level1Obj.parentCc.id+","+level1Obj.externalId+","+level1Obj?.glCodes?.[0]?.value+"\n";
    }
    else{
        //Write logic to create Level 1 Cost Center
        let level1Desc =  project.Description.value;
        let level1ParentID = 60251205046/*Need to Hardocode to the Tree Cost center ID*/;
        let level1Glcode = project.ProjectID.value.substring(0,2)+"-"+project.ProjectID.value.substring(2,5)+"-"+project.ProjectID.value.substring(5,7)+"-"+project.ProjectID.value.substring(7);
        let level1Abbreviation = project.ProjectID.value;
        if(!reportOnly){
            let level1Response = await createCostCenter(level1Name,level1Abbreviation,level1Desc,level1ExternalId,null,level1ParentID,level1Glcode,true,false)
            if(level1Response.items[0].status == "OK"){
                level1ID = level1Response.items[0].id;
                ukgCC.push({"id":level1ID,"name":level1Name});
                logsCreated += level1Name+","+1+","+level1ParentID+","+level1ExternalId+","+level1Glcode+"\n";
            }
            else{
                level1ID = null;
                logsFailed+= level1Name+","+1+","+level1Response.items[0].errors[0].message+"\n";
            }
        }
        else{
            logsCreated += level1Name+","+1+","+level1ParentID+","+level1ExternalId+","+level1Glcode+"\n";
        }
    }
    var contextProjBudget = {};
    contextProjBudget['$filter'] = "ProjectID eq '"+project.ProjectID.value+"'";
    let projectBudgetsObj = acumatica.ProjectBudget_GetList(contextProjBudget);
    let projectBudgets = JSON.parse(projectBudgetsObj.body);
    for(var projBudget of projectBudgets){
        if(projBudget.AccountGroup.value === "LABOR"){
            var contextProjectTask = {};
            contextProjectTask['$filter'] = "ProjectID eq '"+project.ProjectID.value+"'";
            contextProjectTask['$custom'] = "Task.TaskID";
            let projectTaskObj = acumatica.ProjectTask_GetList(contextProjectTask);
            let projectTask = JSON.parse(projectTaskObj.body);
            //Level 2 required details
            let level2Name = projBudget.ProjectTaskID.value+"--"+projectTask.find(pt = pt.ProjectTaskID.value === projBudget.ProjectTaskID.value).Description.value;
            let level2Found = ukgCC.find(cc = cc.name === level2Name) != undefined ? true : false;
            let level2ID = null;
            let level2Obj = null;
            let level2ExternalId = level1ExternalId+"/"+projectTask.find(pt = pt.ProjectTaskID.value === projBudget.ProjectTaskID.value).custom.Task.TaskID.value/*Add the externalID*/;
            let level2Desc =  projectTask.find(pt = pt.ProjectTaskID.value === projBudget.ProjectTaskID.value).Description.value;
            let level2Glcode = projBudget.ProjectTaskID.value;
            let level2Abbreviation = projBudget.ProjectTaskID.value;
            //Level 3 required details
            let level3Name = projBudget.CostCode.value+"--"+projBudget.Description.value;
            let level3ID = null;
            let level3Obj = null;
            let level3ExternalId = null/*Add the externalID*/;
            let level3Desc =  projBudget.Description.value;
            let level3Glcode = projBudget.CostCode.value;
            let level3Abbreviation = projBudget.CostCode.value;
            if(level2Found == true){
                log.info("Level 2 Cost center "+level2Name+"Already Exists Hence not created");
                level2Obj = ukgCC.find(cc = cc.name === level2Name);
                level2ID = level2Obj.id;
                logsIgnored+= level2Name+","+1+","+level2Obj?.parentCc?.id+","+level2Obj.externalId+","+level2Obj?.glCodes?.[0]?.value+"\n";
                let level3Found = ukgCC.find(cc = cc.name === level3Name) != undefined ? true : false;
                if(level3Found == true){
                    log.info("Level 3 Cost center "+level3Name+"Already Exists Hence not created");
                    level3Obj = ukgCC.find(cc = cc.name === level3Name);
                    logsIgnored+= level3Name+","+3+","+level3Obj.parentCc.id+","+level3Obj.externalId+","+level3Obj?.glCodes?.[0]?.value+"\n";
                }
                else{
                    //Write Logic to create level3
                    if(!reportOnly){
                        let level3Response = await createCostCenter(level3Name,level3Abbreviation,level3Desc,level3ExternalId,null,level2ID,level3Glcode,true,true);
                        if(level3Response.items[0].status == "OK"){
                            level3ID = level3Response.items[0].id;
                            logsCreated += level3Name+","+3+","+level2ID+","+level3ExternalId+","+level3Glcode+"\n";
                        }
                        else{
                            level3ID = null;
                            logsFailed+= level3Name+","+3+","+level3Response.items[0].errors[0].message+"\n";
                        }
                    }
                    else{
                        logsCreated += level3Name+","+3+","+level2ID+","+level3ExternalId+","+level3Glcode+"\n";

                    }
                }

            }
            else{
                //Write Logic for creating level 2 CC and level 3 CC;
                if(!reportOnly){
                    let level2Response = await createCostCenter(level2Name,level2Abbreviation,level2Desc,level2ExternalId,null,level1ID,level2Glcode,true,false)
                    if(level2Response.items[0].status == "OK"){
                        level2ID = level2Response.items[0].id;
                        logsCreated += level2Name+","+2+","+level1ID+","+level2ExternalId+","+level2Glcode+"\n";
                        ukgCC.push({"id":level2ID,"name":level2Name});
                        let level3Response = await createCostCenter(level3Name,level3Abbreviation,level3Desc,level3ExternalId,null,level2ID,level3Glcode,true,true);
                        if(level3Response.items[0].status == "OK"){
                            level3ID = level3Response.items[0].id;
                            logsCreated += level3Name+","+3+","+level2ID+","+level3ExternalId+","+level3Glcode+"\n";
                        }
                        else{
                            level3ID = null;
                            logsFailed+= level3Name+","+3+","+level3Response.items[0].errors[0].message+"\n";
                        }
                    }
                    else{
                        level2ID = null;
                        logsFailed+= level2Name+","+2+","+level2Response.items[0].errors[0].message+"\n";
                    }
                }
                else{
                    logsCreated += level2Name+","+2+","+level1ID+","+level2ExternalId+","+level2Glcode+"\n";
                    logsCreated += level3Name+","+3+","+level2ID+","+level3ExternalId+","+level3Glcode+"\n";
                }
                
            }
        }
    }
}
log.info("List of Cost center Created\n"+logsCreated);
log.info("================================================================================================");
log.info("List of Cost center Failed to create\n"+logsFailed);
log.info("================================================================================================");
log.info("List of Cost center Ignored as they already Existed\n"+logsIgnored);
log.info("================================================================================================");

async function createCostCenter(name,abbreviation,description,externalId,parentLevel,parentId,glCode,visible,timeAllocation){
    let createBody = {};
    createBody.companyName = "6204759";
    createBody.cost_centers = {};
    createBody.cost_centers.name = name;
    createBody.cost_centers.abbreviation = abbreviation;
    createBody.cost_centers.description = description;
    createBody.cost_centers.external_id = externalId;
    createBody.cost_centers.parent_cc = {};
    createBody.cost_centers.parent_cc.tree_index = null;
    createBody.cost_centers.parent_cc.id = parentId;
    createBody.cost_centers.gl_codes = [];
    createBody.cost_centers.gl_codes.push({"index": 0, "value": glCode});
    createBody.cost_centers.visible = visible;
    createBody.cost_centers.time_allocation= timeAllocation;
    let newCCObjUKG = ukg.costCentersBatchUpload(createBody);
    let ccUKGResponse = JSON.parse(newCCObjUKG.body);
    log.info("For CC "+name+" Response is: "+JSON.stringify(ccUKGResponse));
    return ccUKGResponse;
}
