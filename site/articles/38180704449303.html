Schema Table Creation

Refer to the following options to automate the creation of schema tables:

Note: You must verify the following points after a table is created:

  • Update the description as necessary.
  • Update as "String Array" or "Integer Array" when multiValued = true.
  • Add a red asterisk for required attributes.
  • Check for uniqueness.
    Notes: 
    • If the uniqueness of any attribute in the schema is server, make sure that we use “unique” term in the description. For example, The unique phone number of the user.
    • In case the  uniqueness is global, we will add the following note in the Description column:
      Note: The value must be unique across all instances of the ABC application.
  • Add canonicalValues if any. For example, Note: The supported values are work and home.
  • Remove all custom fields/attributes and put the format of the custom attribute along with the proper data types. For example, applications[].jobDetails.customFields.<customAttributeName> 
  • Remove rows of complex data types.

OpenAPI Table Creation

Use the following script to automate the creation of OpenAPI tables:

// Helper function to flatten deeply nested properties into an array with full paths and levels
function flattenProperties(properties, parentPath = '', levels = 0, operationId = '', required = false) {
let flattened = [];
for (let prop in properties) {
let fullPath = parentPath ? `${parentPath} > ${prop}` : prop;
let property = properties[prop];
let isRequired = required || (properties.required && properties.required.includes(prop));
flattened.push({ 
name: fullPath, 
level: levels, 
operationId, 
required: isRequired, 
multivalued: property.type === 'array', 
...property 
});
if (property.type === "object" && property.properties) {
flattened.push(...flattenProperties(property.properties, fullPath, levels + 1, operationId, isRequired));
}
if (property.type === "array" && property.items && property.items.properties) {
flattened.push(...flattenProperties(property.items.properties, fullPath, levels + 1, operationId, isRequired));
}
}
return flattened;
}
// Helper function to filter out properties with type "object" or "array"
function filterPropertiesByType(properties) {
return properties.filter(property => property.type !== "object" && property.type !== "array");
}
// Function to construct the visualizer payload
function constructVisualizerPayload() {
var res = pm.response.json();
let operationData = [];
let schemaData = [];
if (res && res.document) {
if (res.document.paths) {
for (let path in res.document.paths) {
let pathItem = res.document.paths[path];
for (let method in pathItem) {
if (['get', 'post', 'put', 'delete', 'patch', 'head', 'options'].includes(method)) {
let operation = pathItem[method];
if (operation && operation.operationId) {
let pathParameters = [];
if (operation.parameters) {
pathParameters = operation.parameters.map(parameter => ({
name: parameter.name,
type: parameter.schema ? parameter.schema.type : '',
description: parameter.description ? parameter.description : '',
level: 0,
operationId: operation.operationId,
required: parameter.required || false,
multivalued: parameter.schema && parameter.schema.type === 'array'
}));
}
let requestBodyParameters = [];
if (operation.requestBody && operation.requestBody.content) {
let content = operation.requestBody.content;
for (let contentType in content) {
let schema = content[contentType].schema;
if (schema && schema.properties) {
let flattenedProperties = flattenProperties(schema.properties, '', 0, operation.operationId);
let filteredProperties = filterPropertiesByType(flattenedProperties);
filteredProperties.sort((a, b) => a.name.localeCompare(b.name));
requestBodyParameters = filteredProperties;
}
}
}
operationData.push({
operationId: operation.operationId,
parameters: pathParameters.concat(requestBodyParameters)
});
}
}
}
}
}
if (res.document.components && res.document.components.schemas) {
for (let schemaName in res.document.components.schemas) {
let schema = res.document.components.schemas[schemaName];
let flattenedSchemaProperties = flattenProperties(schema.properties);
let filteredSchemaData = filterPropertiesByType(flattenedSchemaProperties);
schemaData = schemaData.concat(filteredSchemaData.map(property => ({
schemaName: schemaName, // This should be the schema name
operationId: schemaName, // Use schemaName as Operation ID for schema data
propertyName: property.name,
type: property.type,
description: property.description,
required: schema.required && schema.required.includes(property.name.split(' > ').pop()) ? true : property.required || false,
multivalued: property.multivalued || false
})));
}
}
}
return { response: { operationData, schemaData } };
}

OpenAPI Documentation Process

  1. In your draft article, under the OpenAPI section, upload your API file.
  2. Attach your OpenAPI (Swagger) YAML or JSON file to your draft article in Zendesk Guide.
  3. Publish your document internally, set the article visibility to internal and publish it.
    This step generates a public/internal link for your attachment.
  4. Copy your OpenAPI spec link, and open the published draft article.
  5. Right-click the attachment link for your OpenAPI file and copy the link address
  6. The link should look something like: https://support.yoursite.com/hc/en-us/article_attachments/1234567890/your_api.yaml/Json
  7. Copy the Swagger rendering code from the Content Enhancements - Development page.
    Find the Swagger rendering code sample.
  8. Paste the code in your draft document. In your Zendesk article editor, switch to Source Code view (usually a <></> button)
  9. Paste the copied Swagger rendering code where you want the API docs to appear.

INFO ABOUT CODE BLOCK
You might face problems with code blocks as Zendesk or any static html website generator by default converts any HTML code. For example, If angle braces are present in your code, your code block does not render, or the block breaks. In this case, the following steps will help you to fix the problem:

  1. Verify if your code has special characters such as <, >, &.
  2. Copy your code segment.
  3. Navigate to https://codebeautify.org/html-encode-string
  4. Paste your code under **Enter the text to be HTML encoded** textbox.
  5. Click **HTML Encode**.
  6. Click **Copy to Clipboard**.
  7. Paste this code within the code block and save it.
